[docs]def spiral(t, x_range=100, y_range=10):
    """Make a spiral."""
    for x in range(x_range):
        for y in range(y_range):
            t.forward(y / 4)
        t.left(x * 2)




[docs]def recursive_star(t, order, size, angles=[60, -120, 60, 120]):
    """Make a recursive, fractal-esque star."""
    if order is 0:
        t.forward(size)
    else:
        for angle in angles:
            recursive_star(t, order - 1, size / 3)
            t.left(angle)

[docs]def prog__schizophrenic_spiral(t, max_steps=100, size=10):
    def basic(t, max_range):
        for n in range(max_steps):
            t.forward(random.randrange(0, size + n + 1))
            t.left(random.randrange(0, size + n + 1))
            t.left(random.randrange(0, size + n + 1))
            t.left(random.randrange(-size + n + 1, size + n + 1))
            if n % 3 == 0:
                t.forward(random.randrange(0, size + n + 1))

    for step in range(max_steps):
        basic(t, step % 2 + 1)
